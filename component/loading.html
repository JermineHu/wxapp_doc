loading

​消息提示框

属性名 类型 默认值 说明
hidden Boolean false 是否隐藏

示例代码：

<view class="body-view">
    <loading hidden="{{hidden}}">
        加载中...
    </loading>
    <button type="default" bindtap="loadingTap">点击弹出loading</button>
</view>
Page({
    data: {
        hidden: true
    },
    loadingTap: function(){
        this.setData({
          hidden: false
        });
        var that = this;
        setTimeout(function(){
          that.setData({
              hidden: true
          });
          that.update();
        }, 3000);
    }
})
